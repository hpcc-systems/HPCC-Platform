set(SRC
    esdl.bat.in
    )

foreach(source ${SRC})
    string(REGEX REPLACE "\\.in$" "" configured_source ${source})
    configure_file(${source} ${configured_source} @ONLY)
    install(PROGRAMS ${CMAKE_CURRENT_BINARY_DIR}/${configured_source} DESTINATION IDEPlugins/ESDL)
endforeach()
