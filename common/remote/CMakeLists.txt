         rmtfile.cpp 
         rmtpass.cpp 
         rmtspawn.cpp 
         rmtssh.cpp 
         rmtsmtp.cpp 
         sockfile.cpp
         
         remoteerr.hpp
         rmtfile.hpp

         rmtspawn.hpp
         rmtssh.hpp
         sockfile.hpp
    )

  set ( SRCS

         ./../../system/hrpc 
         ./../../system/mp 
         ./../../system/include 
         ./../../system/jlib 
         ./../../system/jhtree
         ./../../rtl/eclrtl
         ./../../system/security/securesocket
         ./../../testing/unittests
         ./../../rtl/include

    jhtree 
    mp
    ${URIPARSER_LIBRARIES}
    ${CPPUNIT_LIBRARIES}
    )

IF (USE_OPENSSL)
    target_link_libraries ( remote 
    	securesocket
